  constructor() {
    this.first = null;
    this.last = null;
    this.size = 0;

  enqueue(val) {

      this.first = newNode;

      this.last.next = newNode;
      this.last = newNode;

    if (this.first === this.last) {
      this.last = null;
    }
    this.first = this.first.next;
    this.size--;
    return temp.val;
  }
}
